		\parbox{\linewidth}{
			\begin{minipage}{0.2\linewidth}
				\qrcode{#1}
			\end{minipage} \hfill
			\begin{minipage}{0.8\linewidth}
				\textbf{#2} \\ #1
			\end{minipage}
	}}}
